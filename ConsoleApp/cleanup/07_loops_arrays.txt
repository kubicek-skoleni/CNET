
//for loop
for (int i = 1; i <= 10; i = i + 2)
{
    Console.Write($"{i} ");
}

//while loop
Console.Write("While loop (countdown): ");
int countdown = 5;
while (countdown > 0)
{
    Console.Write($"{countdown} ");
    countdown--;
}
