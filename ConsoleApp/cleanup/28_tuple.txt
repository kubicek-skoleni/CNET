(string den, bool pracovni) DayOfWeekInCzech(int dayNumber)
{
    string den = dayNumber switch
    {
        1 => "Pondìlí",
        2 => "Úterý",
        3 => "Støeda",
        4 => "Ètvrtek",
        5 => "Pátek",
        6 => "Sobota",
        7 => "Nedìle",
        _ => "Neplatný den"
    };

    bool jePracovniDen = (dayNumber >= 1 && dayNumber <= 5);
        
    return (den, jePracovniDen);
}
